class Ship {
  constructor(length) {
    this.length = length;
    this.hits = 0;
    this.sunk = false;
  }
  
  hit() {
    this.hits++;
    if (this.hits >= this.length) {
      this.sunk = true;
    }
  }
}

function calculate(n1, operator, n2) {
  let result = '';
  
  if (operator === 'add') {
    result = parseFloat(n1) + parseFloat(n2);
  } else if (operator === 'subtract') {
    result = parseFloat(n1) - parseFloat(n2);
  }
  
  return result;
}
